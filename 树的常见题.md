# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def preorderTraversal(self, root: TreeNode) -> List[int]:
        #dfs 
    #     if not root: return []
    #     res = []
    #     self.dfs(root,res)

    #     res.append(root.val)
    #     if root.left: self.dfs(root.left, res)
    #     if root.right: self.dfs(root.right, res)
        #bfs 根左右
        if not root: return []
        stack, res = [root], []

#### 二叉树的直径
```python
# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def diameterOfBinaryTree(self, root: TreeNode) -> int:
        self.dia = 0
        def dfs(root):#获取左右子树的深度
            if not root: return 0
            depth1 = dfs(root.left)
            depth2 = dfs(root.right)
            self.dia = max(self.dia, depth1 + depth2)
            return max(depth1,depth2) + 1
        dfs(root)
        return self.dia
            
```

#### 从前序遍历和中序遍历构造二叉树

# Definition for a binary tree node.
# class TreeNode:

#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def buildTree(self, pre: List[int], inorder: List[int]) -> TreeNode:
        if not pre or not inorder: return None
        root = TreeNode(pre[0])
        root_index = inorder.index(root.val)
        root.left = self.buildTree(pre[1:root_index+1],inorder[:root_index])
        root.right = self.buildTree(pre[root_index+1:],inorder[root_index+1:])
        return root

#### 二叉树的最近祖先节点
```python
# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def lowestCommonAncestor(self, root: 'TreeNode', p: 'TreeNode', q: 'TreeNode') -> 'TreeNode':
        # p。q要么在一个节点的左右子树上，要么是有一个在根节点上
        if not root or p==root or q == root: return root#首先判断p，q是否有父(孙)子关系
        left = self.lowestCommonAncestor(root.left,p,q)
        right = self.lowestCommonAncestor(root.right,p,q)
        if left and right:
            return root
        if left: return left
        if right: return right
```

#### 是否平衡二叉树
```python
class Solution:
    def isBalanced(self, root: TreeNode) -> bool:
        def dfs(root):
            if not root: return 0
            left_height = dfs(root.left)
            right_height = dfs(root.right)
            if left_height == -1 or right_height == -1 or abs(left_height - right_height)>1:
                return -1
            else:
                return max(left_height,right_height)+1
        return dfs(root)>=0
```

#### 将二叉树展开为右侧单链
```python
# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def flatten(self, root: TreeNode) -> None:
        if not root: return root
        self.flatten(root.left)#将左树展开成单链
        self.flatten(root.right)#将右树展开成单链
        left = root.left
        if not left: return #递归终止条件，如果左右子树不能被展开为链表了
        while left.right:#找到左树(链表的最后一个节点)
            left = left.right
        left.right = root.right#末尾节点连接右树
        root.right = root.left#根左移动到根右
        root.left = None
```
